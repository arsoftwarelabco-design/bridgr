async function analyzeRepo(repoUrl) {
  // 1. Show loading state
  showLoadingState();
  
  // 2. Fetch repo data from GitHub public API (no key needed)
  const repoPath = repoUrl.replace("https://github.com/","")
                          .replace(/\/$/, "");
  const meta = await fetch(
    `https://api.github.com/repos/${repoPath}`
  ).then(r => r.json());
  
  const readmeData = await fetch(
    `https://api.github.com/repos/${repoPath}/readme`
  ).then(r => r.json()).catch(() => ({ content: "" }));
  
  const readmeContent = readmeData.content 
    ? atob(readmeData.content.replace(/\n/g,"")) 
    : "No README available";

  // 3. Call our local proxy (no CORS issues)
  const res = await fetch("/api/analyze", {
    method: "POST",
    headers: { "Content-Type": "application/json" },
    body: JSON.stringify({ repoUrl, readmeContent, repoMeta: meta })
  });
  const data = await res.json();
  
  if (data.success) {
    renderReport(data.report, meta);
  } else {
    showError(data.error);
  }
}

        // DOM elements
        const inputPanel = document.getElementById('inputPanel');

        const repoInput = document.getElementById('repoInput');

        const chatMessages = document.getElementById('chatMessages');

            if (repoUrl) {
                analyzeRepo(repoUrl);
            }
        });

        // Enter key handler for repo input
        repoInput.addEventListener('keypress', (e) => {
            if (e.key === 'Enter') {
                const repoUrl = repoInput.value.trim();
                if (repoUrl) {
                    analyzeRepo(repoUrl);
                }
            }
        });

        // Chat handlers
        sendButton.addEventListener('click', sendChatMessage);
        chatInput.addEventListener('keypress', (e) => {

        });

        // Show loading state
        function showLoadingState() {
            inputPanel.classList.add('hidden');
            reportPanel.classList.remove('active');
            loadingPanel.classList.add('active');
        }

        // Show error
        function showError(message) {
            loadingPanel.classList.remove('active');
            inputPanel.classList.remove('hidden');
            
            const errorCard = document.createElement('div');
            errorCard.className = 'error-card';
            errorCard.innerHTML = `
                <span class="card-icon">⚠️</span>
                <h3 class="card-title">Error</h3>
                <div class="card-content">${message}</div>
            `;
            
            inputPanel.parentNode.insertBefore(errorCard, inputPanel);
            
            setTimeout(() => {
                errorCard.remove();
            }, 5000);
        }

        // Main analyze function
        async function analyzeRepo(repoUrl) {
            try {
                showLoadingState();

                // Extract repo path
                const repoPath = repoUrl.replace('https://github.com/', '').replace(/\/$/, '');

                // Fetch repo metadata from GitHub
                const metaResponse = await fetch(`https://api.github.com/repos/${repoPath}`);
                const meta = await metaResponse.json();

                if (meta.message === 'Not Found') {
                    showError('Repository not found. Make sure it\'s a public GitHub repository.');
                    return;
                }

                // Fetch README
                let readmeContent = 'No README available';
                try {
                    const readmeResponse = await fetch(`https://api.github.com/repos/${repoPath}/readme`);
                    const readmeData = await readmeResponse.json();
                    if (readmeData.content) {
                        readmeContent = atob(readmeData.content.replace(/\n/g, ''));
                    }
                } catch (e) {
                    console.log('No README found');
                }

                // Call our local proxy
                const response = await fetch('/api/analyze', {
                    method: 'POST',
                    headers: { 'Content-Type': 'application/json' },
                    body: JSON.stringify({ repoUrl, readmeContent, repoMeta: meta })
                });

                const data = await response.json();

                if (data.success) {
                    currentReport = data.report;
